/**
 * Definition for binary tree
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool isBalanced(TreeNode *root) {
        // Start typing your C/C++ solution below
        // DO NOT write int main() function
        if (root == NULL)
            return true;
        bool ret = true;
        b(root, ret);
        return ret;
        
    }
    int b(TreeNode* root, bool &ret) {
        if (root == NULL)
            return 0;
        else {
            int l = b(root->left, ret);
            int r = b(root->right, ret);
            if (abs(l-r) > 1)
                ret = false;
            return (l > r)?(l+1):(r+1);
        }
            
    }
};
