class Solution {
public:
    int divide(int dividend, int divisor) {
        // Start typing your C/C++ solution below
        // DO NOT write int main() function
        bool pos = true;
        double upper = dividend, lower = divisor;
        if (dividend < 0) {
            pos = !pos;
            upper = -upper;
        }
        if (divisor < 0) {
            pos = !pos;
            lower = -lower;
        }
        double ret = exp(log(upper) - log(lower));
        return pos?ret:-ret;
    }
};
