/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode *partition(ListNode *head, int x) {
        // Start typing your C/C++ solution below
        // DO NOT write int main() function
        if (head == NULL)
            return head;
        ListNode* l = new ListNode(0);
        ListNode* lt = l;
        ListNode* r = new ListNode(0);
        ListNode* rt = r;
        while (head != NULL) {
            if (head->val >= x) {
                rt->next = head;
                rt = rt->next;
            }
            else if (head->val < x) {
                lt->next = head;
                lt = lt->next;
            }
            ListNode* temp = head;
            head = head->next;
            temp->next = NULL;
        }
        lt->next = r->next;
        delete r;
        head = l->next;
        delete l;
        return head;
    }
};
