class Solution {
public:
    vector<int> getRow(int rowIndex) {
        // Start typing your C/C++ solution below
        // DO NOT write int main() function
        vector<int> ret(rowIndex+1, 0);
        ret[0] = 1;
        for (int j = 1; j <= rowIndex; j++)
            ret[j] = ((double) ret[j-1] * (rowIndex-j+1))/j;
        return ret;
        
    }
};
