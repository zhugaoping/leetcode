class Solution {
public:
    vector<int> getRow(int rowIndex) {
        // Start typing your C/C++ solution below
        // DO NOT write int main() function
        vector<int> ret;
        ret.push_back(1);
        if (rowIndex > 0) {
            for (int i = 1; i <= rowIndex; i++)
                ret.push_back(((double)ret.back())*(rowIndex - i + 1)/i);
        }
        return ret;
    }
};
