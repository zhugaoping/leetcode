class Solution {
public:
    int maxProfit(vector<int> &prices) {
        // Start typing your C/C++ solution below
        // DO NOT write int main() function
        int size = prices.size();
        if (size < 2)
            return 0;
        int minp = prices[0];
        int bene = 0;
        for (int i = 1; i < size; i++) {
            bene = max((prices[i] - minp), bene);
            minp = (prices[i] < minp)?prices[i]:minp;
        }
        return bene;
    }
};
