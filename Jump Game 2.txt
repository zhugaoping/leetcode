class Solution {
public:
    int jump(int A[], int n) {
        // Start typing your C/C++ solution below
        // DO NOT write int main() function
        int oldLast = 0, newLast = INT_MIN, step = 0, i = 0;
        while (i <= min(n-1, oldLast)) {
            newLast = max(newLast, i + A[i]);
            if (i == oldLast) {
                step += (i == n - 1)?0:1;
                oldLast = newLast;
            }
            i++;
        }
        return step;
    }
};
