class Solution {
public:
    int jump(int A[], int n) {
        // Start typing your C/C++ solution below
        // DO NOT write int main() function
        int s = 0, i = 0, t = 0;
        while (i < n) {
            int ot = t;
            for (i; i <= ot; i++)
                t = ((i + A[i]) > t)?(i + A[i]):t;
            s++;
            if (t >= n - 1)
                break;
        }
        return s;
    }
};
