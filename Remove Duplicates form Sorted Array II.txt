class Solution {
public:
    int removeDuplicates(int A[], int n) {
        // Start typing your C/C++ solution below
        // DO NOT write int main() function
        if (n == 0 || n == 1)
            return n;
        int e = 1;
        for (int i = 2; i < n; i++)
            if (A[i] != A[e-1])
                A[++e] = A[i];
        return e+1;
    }
};
