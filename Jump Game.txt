class Solution {
public:
    bool canJump(int A[], int n) {
        // Start typing your C/C++ solution below
        // DO NOT write int main() function
        int mav = A[0];
        for (int i = 0; ((i < n) && (i <= mav)); i++)
            mav = max(mav, i + A[i]);
        return (mav >= (n-1))?true:false;
        
    }
};
