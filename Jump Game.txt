class Solution {
public:
    bool canJump(int A[], int n) {
        // Start typing your C/C++ solution below
        // DO NOT write int main() function
        if (n == 0)
            return true;
        int last = A[0];
        for (int i = 1; i <= min(last, n-1); i++)
            last = max(last, i + A[i]);
        return (last >= n - 1);
    }
};
