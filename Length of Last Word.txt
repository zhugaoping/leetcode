class Solution {
public:
    int lengthOfLastWord(const char *s) {
        // Start typing your C/C++ solution below
        // DO NOT write int main() function
        int size = strlen(s), l = size - 1;
        while ((l >= 0) && !isalpha(tolower(*(s+l))))
            l--;
        if (l < 0)
            return 0;
        int u = l;
        while ((l >= 0) && isalpha(tolower(*(s+l))))
            l--;
        return u - l;
    }
};
