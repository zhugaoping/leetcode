class Solution {
public:
    TreeNode* next;
    void flatten(TreeNode *root) {

        next = NULL;
        _flatten(root);
    }
    void _flatten(TreeNode* root) {
        if (root == NULL)
            return;
        _flatten(root->right);
        _flatten(root->left);
        if (next != NULL)
            root->right = next;
        root->left=next;
        next = root;
        next->left = NULL;
    }
