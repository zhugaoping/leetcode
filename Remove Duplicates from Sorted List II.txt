/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode *deleteDuplicates(ListNode *head) {
        // Start typing your C/C++ solution below
        // DO NOT write int main() function
        if (head == NULL)
            return head;
        ListNode* h = new ListNode(0);
        ListNode* t = h, *p;
        while (head != NULL) {
            p = head;
            while ((p != NULL) && (p->val == head->val))
                p = p->next;
            if (p == head->next) {
                t->next = head;
                t = head;
                head = p;
            }
            else {
                while (head != p) {
                    ListNode* r = head;
                    head = head->next;
                    delete r;
                }   
            }
        }
        t->next = NULL;
        head = h->next;
        delete h;
        return head;
        
    }
};
