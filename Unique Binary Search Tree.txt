class Solution {
public:
    int numTrees(int n) {
        // Start typing your C/C++ solution below
        // DO NOT write int main() function
        if (n < 2)
            return 1;
        double plan[n+1];
        plan[0] = 1;
        plan[1] = 1;
        for (int i = 2; i <= n; i++) {
            plan[i] = 0; 
            for (int j = 0; j < i; j++) {
                plan[i] += plan[j]*plan[i-1-j];
            }
        }
        return plan[n];
    }
};
