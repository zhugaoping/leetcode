class Solution {
public:
    bool isPalindrome(string s) {
        // Start typing your C/C++ solution below
        // DO NOT write int main() function
        if (s == "")
            return true;
        int l = 0, u = s.size() - 1;
        while ((l < u) && !(isalpha(s[l]) || isdigit(s[l])))
            l++;
        while ((l < u) && !(isalpha(s[u]) || isdigit(s[u])))
            u--;
        while (l < u) {
            if (tolower(s[l]) != tolower(s[u]))
                return false;
            l++;
            u--;
            while ((l < u) && !(isalpha(s[l]) || isdigit(s[l])))
                l++;
            while ((l < u) && !(isalpha(s[u]) || isdigit(s[u])))
                u--;
        }
        return true;
    }
};
