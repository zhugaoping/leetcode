class Solution {
public:
    vector<int> plusOne(vector<int> &digits) {
        // Start typing your C/C++ solution below
        // DO NOT write int main() function
        int i;
        for (i = digits.size() - 1; i >= 0; i--) {
            if (++digits[i] >= 10)
                digits[i] %= 10;
            else
                break;
        }
        if (i == -1)
            digits.insert(digits.begin(), 1);
        return digits;
    }
};
