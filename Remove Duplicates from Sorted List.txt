/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode *deleteDuplicates(ListNode *head) {
        // Start typing your C/C++ solution below
        // DO NOT write int main() function
        if (head == NULL)
            return head;
        ListNode* t = head;
        ListNode* n = head->next;
        while (n != NULL) {
            if (n->val != t->val) {
                t->next = n;
                t = n;
                n = n->next;
            }
            else {
                ListNode* p = n;
                n = n->next;
                delete p;
            }
        }
        t->next = NULL;
        return head;
        
    }
};
