class Solution {
public:
    bool isPalindrome(int x) {
        // Start typing your C/C++ solution below
        // DO NOT write int main() function
        if (x < 0)
            return false;
        int div = 1;
        while (x/div >= 10)
            div *= 10;
        while (x >= 10) {
            if (x/div != x%10)
                return false;
            x = (x % div)/10;
            div = div/100;
        }
        return true;
    }
};
